NAME

Ten.Highlight - Highlight texts in the document.

SYNOPSIS

var h = new Ten.Highlight('highlighted');
h.show();

var BlueHighlight = new Ten.Class({
    base: [Ten.Highlight],
    Color: new Ten.Color(0,0,255,0.3)
});
var b = new BlueHighlight('highlighted');
b.show();
